function pushbutton1_Callback(hObject, eventdata, handles)
% hObject    handle to pushbutton1 (see GCBO)
% eventdata  reserved - to be defined in a future version of MATLAB
% handles    structure with handles and user data (see GUIDATA)

global file_name
global Sound_results

file_name = get(handles.edit1,'String');
freq = str2double(get(handles.edit2,'String'));
tmp = get(handles.checkbox1,'Value');
if(tmp == 1)
    x = wavread(file_name);
else
    x = zeros(2000000,1);
end

tmp2 = get(handles.checkbox2,'Value');

tmp3 = get(handles.edit3,'string');
tmp3 = tmp3 - 'a';
if(tmp2 == 1)
    Sound_results = SoundGenerate(x,freq,tmp3);
else
    Sound_results = x;
end

sound(Sound_results,44100);
figure;
plot(Sound_results);

end_msg = 'END';
end_msg
